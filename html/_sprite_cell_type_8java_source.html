355 public static Image scale(Image src, int newW, int newH) {
356 BufferedImage tmp = (BufferedImage) src;
357 int srcW = tmp.getWidth();
358 int srcH = tmp.getHeight();
359 BufferedImage dst = new BufferedImage(newW, newH, tmp.getType());
360
361
362 double xRatio = (double) srcW / newW;
363 double yRatio = (double) srcH / newH;
364
365 for (int y = 0; y < newH; y++) {
366 int srcY = (int) (y * yRatio);
367 if (srcY >= srcH) srcY = srcH - 1;
368
369 for (int x = 0; x < newW; x++) {
370 int srcX = (int) (x * xRatio);
371 if (srcX >= srcW) srcX = srcW - 1;
372 int rgb = tmp.getRGB(srcX, srcY);
373 dst.setRGB(x, y, rgb);
374 }
375 }
376 return (Image)dst;
377 }
